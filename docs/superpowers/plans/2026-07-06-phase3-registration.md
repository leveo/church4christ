# Phase 3: Registration Module — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A Supabase-only Registration module: admins build events with custom question forms and optional Stripe payment; anyone (guest or member) registers; capacity enforced; answers viewable/exportable.

**Architecture:** `reg_events` (+i18n) own questions (`reg_questions` + i18n, typed, ordered) and registrations (+`reg_answers`). Free events confirm instantly; paid events create a `pending` registration + Stripe Checkout session (reusing Phase 2's `src/lib/stripe.ts` and the shared `/api/stripe/webhook` endpoint — a `metadata.kind = 'registration'` branch), confirmed by `checkout.session.completed`, released by `checkout.session.expired`. Logic in `src/lib/regDb.ts`; thin pages.

**Tech Stack:** Same as Phase 2.

**Spec:** `docs/superpowers/specs/2026-07-06-supabase-giving-registration-design.md`
**Depends on:** Phases 1–2 (AppDb, module gating with the `registration` registry entry + route policy prefixes already landed, `stripe.ts`, webhook endpoint).

## Global Constraints

Same as Phase 2 (Supabase-only gating, integer cents, mocked Stripe in tests, i18n parity, portable SQL style, all suites green per task, commit per task).

```sql
-- Registration module schema (Supabase-only module).
CREATE TABLE reg_events (
  id integer PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  starts_at TEXT NOT NULL,          -- YYYY-MM-DD HH:MM:SS (UTC, like the rest of the app)
  ends_at TEXT,
  location TEXT,
  capacity INTEGER CHECK (capacity IS NULL OR capacity > 0),   -- NULL = unlimited
  price_cents INTEGER CHECK (price_cents IS NULL OR price_cents >= 0), -- NULL/0 = free
  currency TEXT NOT NULL DEFAULT 'usd',
  opens_at TEXT,                    -- NULL = open immediately
  closes_at TEXT,                   -- NULL = open until start
  active INTEGER NOT NULL DEFAULT 1,
  created_at TEXT NOT NULL DEFAULT (datetime('now')),
  updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);

CREATE TABLE reg_event_i18n (
  event_id integer NOT NULL REFERENCES reg_events(id) ON DELETE CASCADE,
  locale TEXT NOT NULL CHECK (locale IN ('en','zh')),
  title TEXT NOT NULL,
  description TEXT,
  PRIMARY KEY (event_id, locale)
);

CREATE TABLE reg_questions (
  id integer PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  event_id integer NOT NULL REFERENCES reg_events(id) ON DELETE CASCADE,
  sort INTEGER NOT NULL DEFAULT 0,
  type TEXT NOT NULL CHECK (type IN ('text','textarea','select','checkbox','yesno')),
  required INTEGER NOT NULL DEFAULT 0,
  options TEXT                       -- JSON array of strings, select/checkbox only
);

CREATE TABLE reg_question_i18n (
  question_id integer NOT NULL REFERENCES reg_questions(id) ON DELETE CASCADE,
  locale TEXT NOT NULL CHECK (locale IN ('en','zh')),
  label TEXT NOT NULL,
  PRIMARY KEY (question_id, locale)
);

CREATE TABLE registrations (
  id integer PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  event_id integer NOT NULL REFERENCES reg_events(id) ON DELETE CASCADE,
  person_id integer REFERENCES people(id) ON DELETE SET NULL,
  name TEXT NOT NULL,
  email TEXT NOT NULL,
  status TEXT NOT NULL DEFAULT 'pending'
    CHECK (status IN ('pending','confirmed','cancelled')),
  amount_cents INTEGER NOT NULL DEFAULT 0,
  currency TEXT NOT NULL DEFAULT 'usd',
  stripe_checkout_session_id TEXT,
  stripe_payment_intent_id TEXT,
  created_at TEXT NOT NULL DEFAULT (datetime('now')),
  updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE UNIQUE INDEX registrations_session ON registrations(stripe_checkout_session_id)
  WHERE stripe_checkout_session_id IS NOT NULL;
CREATE INDEX registrations_event ON registrations(event_id, status);

CREATE TABLE reg_answers (
  registration_id integer NOT NULL REFERENCES registrations(id) ON DELETE CASCADE,
  question_id integer NOT NULL REFERENCES reg_questions(id) ON DELETE CASCADE,
  value TEXT NOT NULL,               -- checkbox answers JSON-encoded array
  PRIMARY KEY (registration_id, question_id)
);
```

**Interfaces (Produces):**

```ts
export interface RegQuestion { id: number; sort: number; type: 'text'|'textarea'|'select'|'checkbox'|'yesno'; required: number; options: string[] | null; label: string }
export interface RegEvent { id: number; title: string; description: string | null; starts_at: string; ends_at: string | null; location: string | null; capacity: number | null; price_cents: number | null; currency: string; opens_at: string | null; closes_at: string | null; active: number; confirmed_count: number; taken_count: number }

export function listOpenEvents(db: AppDb, locale: Locale): Promise<RegEvent[]>;
// open = active AND (opens_at IS NULL OR opens_at <= datetime('now'))
//        AND (COALESCE(closes_at, starts_at) > datetime('now'))
// taken_count counts status IN ('pending','confirmed') — pending holds a seat
// until Checkout expires. isFull(e) = capacity !== null && taken_count >= capacity.
export function getOpenEvent(db: AppDb, locale: Locale, id: number): Promise<RegEvent | null>;
export function listQuestions(db: AppDb, locale: Locale, eventId: number): Promise<RegQuestion[]>;

/** Validate answers against the event's questions; returns normalized
 *  [questionId, value] pairs or throws 'missing_required' | 'bad_answer'. */
export function validateAnswers(questions: RegQuestion[], form: Record<string, string | string[]>): Array<[number, string]>;
/** Create registration (+answers) atomically via batch(); status per args.
 *  Re-checks capacity inside the same transaction and throws 'event_full'. */
export function createRegistration(db: AppDb, input: { eventId: number; personId: number | null; name: string; email: string; status: 'pending' | 'confirmed'; amountCents: number; currency: string; answers: Array<[number, string]> }): Promise<number>;
export function attachCheckoutSession(db: AppDb, registrationId: number, sessionId: string): Promise<void>;
export function confirmBySession(db: AppDb, sessionId: string, paymentIntentId: string | null): Promise<boolean>; // pending → confirmed, idempotent
export function cancelBySession(db: AppDb, sessionId: string): Promise<boolean>;   // pending → cancelled (expiry)
export function cancelRegistration(db: AppDb, id: number): Promise<boolean>;        // admin action, any status → cancelled

// Admin:
export function listAllEvents(db: AppDb, locale: Locale): Promise<RegEvent[]>; // active + inactive, with counts
export function saveEvent(db: AppDb, input: { id?: number; title_en: string; title_zh: string; description_en?: string; description_zh?: string; starts_at: string; ends_at?: string | null; location?: string | null; capacity?: number | null; price_cents?: number | null; opens_at?: string | null; closes_at?: string | null; active: number }): Promise<number>;
export function saveQuestions(db: AppDb, eventId: number, questions: Array<{ id?: number; sort: number; type: RegQuestion['type']; required: number; options?: string[] | null; label_en: string; label_zh: string }>): Promise<void>; // replace-all in one batch: delete removed ids, upsert the rest
export function listRegistrations(db: AppDb, locale: Locale, eventId: number): Promise<Array<{ id: number; name: string; email: string; status: string; amount_cents: number; created_at: string; answers: Array<{ label: string; value: string }> }>>;
export function registrationsCsv(db: AppDb, locale: Locale, eventId: number): Promise<string>; // header: Name,Email,Status,Amount,Registered + one column per question label; RFC4180 quoting
```

Capacity race note: `createRegistration` runs `SELECT` count + `INSERT` inside `batch()`. On Postgres that is one transaction but not serializable; acceptable for church scale — but still enforce a hard backstop by re-counting AFTER insert in the same batch result and throwing (rolling back) when over capacity: include statement `SELECT count(*) AS c FROM registrations WHERE event_id = ? AND status IN ('pending','confirmed')` as the batch's last statement; check `c > capacity` on the result and compensate with `cancelRegistration` + throw `'event_full'` (a plain-code check after the batch — document it in a comment).

- [ ] **Step 1:** Failing tests: openness windows (opens_at future → hidden; closes_at past → hidden; NULL closes_at + started event → hidden), capacity counting (pending holds a seat; cancelled frees it), full event → `'event_full'`, `validateAnswers` (required missing, select value not in options, checkbox multi-value JSON round-trip, yesno normalized to 'yes'/'no'), confirm/cancel by session idempotency, `saveQuestions` replace-all keeps answers of surviving questions, CSV quoting (value with comma + quote).
- [ ] **Step 2:** Implement. Run — PASS. Commit: `feat(registration): data module`
